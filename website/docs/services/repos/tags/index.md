```sql
SELECT
name,
node_id,
commit,
tarball_url,
zipball_url
FROM github.repos.tags
WHERE owner = '{{ owner }}' -- required
AND repo = '{{ repo }}' -- required
AND per_page = '{{ per_page }}'
AND page = '{{ page }}'
;
```
